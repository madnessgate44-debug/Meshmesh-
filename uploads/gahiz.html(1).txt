        if (conf < 25 && hes > 40) { wallIndicator.textContent = '🔴 WALL DETECTED — PRESSURE COLLAPSE'; wallIndicator.className = 'wall-indicator wall-detected'; }

    function analyzeAnswer(text) {
        const wordCount = text.split(/\s+/).filter(w => w.length > 0).length;
        const fillerPattern = /\b(um|uh|er|ah|like|you\s*know|I\s*mean|sort\s*of|kind\s*of|basically|actually|literally|honestly|so\s*yeah|right\?)\b/gi;
        const fillerMatches = text.match(fillerPattern) || [];
        const fillerCount = fillerMatches.length;
        const hesitationMarkers = (text.match(/\.{3,}|—{1,}|\.\.\.\.|\,{2,}/g) || []).length;
        const totalWords = Math.max(wordCount, 1);
        const fillerPercent = Math.min(100, (fillerCount / totalWords) * 100);
        const hesitationPercent = Math.min(100, ((hesitationMarkers + fillerCount) / totalWords) * 100);
        const timeRatio = timerSeconds / 45;
        const pressureScore = Math.min(100, Math.round((1 - timeRatio) * 70 + fillerPercent * 0.6 + hesitationPercent * 0.5));
        const confidenceScore = Math.max(5, Math.round(100 - pressureScore - fillerPercent * 0.5 - (wordCount < 15 ? 35 : wordCount < 30 ? 15 : 0)));
        return { wordCount, fillerCount, fillerPercent: Math.round(fillerPercent), hesitationPercent: Math.round(hesitationPercent), pressureScore: Math.round(pressureScore), confidenceScore: Math.round(confidenceScore) };
    }

        drawGraph();
    }

    }

    }

        currentQuestionIndex++;
        if (currentQuestionIndex >= questions.length) { endSession(); }
        else { displayQuestion(); answerInput.value = ''; timerSeconds = 45; updateTimerDisplay(); timerDisplay.classList.remove('warning', 'danger'); if (isRunning && !timerInterval) { startTimer(); } }
    }

        isRunning = false;
        if (timerInterval) { clearInterval(timerInterval); timerInterval = null; }
