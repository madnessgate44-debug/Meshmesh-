3. SIMULATION ENGINE (JAVASCRIPT)
------------------------------------------------------------

const simulationEngine = {
  state: "idle",

  userProfile: {
    confidence: 100,
    stress: 0,
    wallPosition: null,
    recoverySpeed: 0
  },

  questions: [
    "Tell me about yourself",
    "Why should we hire you?",
    "Describe a difficult situation",
    "Unexpected follow-up question",
    "Rapid-fire round"
  ],

  currentIndex: 0,

  start() {
    this.state = "active";
    this.runQuestion();
  },

  runQuestion() {
    const pressure = Math.floor(Math.random() * 40 + 20);

    this.userProfile.confidence -= pressure * 0.6;
    this.userProfile.stress += pressure;

    if (this.userProfile.confidence < 45) {
      this.triggerWall();
    }

    this.currentIndex++;
  },

  triggerWall() {
    this.state = "wall_detected";
    this.userProfile.wallPosition = this.currentIndex;
    console.log("WALL DETECTED at", this.currentIndex);
  },

  finish() {
    this.state = "evaluation";
    return {
      wall: this.userProfile.wallPosition,
      confidence: this.userProfile.confidence
    };
  }
};
